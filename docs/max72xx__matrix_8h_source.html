187  void drawPixel(int16_t xx, int16_t yy, uint16_t color) {
188  if (bitmap == nullptr) {
189  return;
190  }
191 
192  // Operating in bytes is faster and takes less code to run. We don't
193  // need values above 200, so switch from 16 bit ints to 8 bit unsigned
194  // ints (bytes).
195  // Keep xx as int16_t so fix 16 panel limit
196  int16_t x = xx;
197  uint8_t y = yy;
198  uint8_t tmp;
199 
200  if (rotation) {
201  // Implement Adafruit's rotation.
202  if (rotation >= 2) {
203  // rotation == 2 || rotation == 3
204  x = _width - 1 - x;
205  }
206 
207  if (rotation == 1 || rotation == 2) {
208  // rotation == 1 || rotation == 2
209  y = _height - 1 - y;
210  }
211 
212  if (rotation & 1) {
213  // rotation == 1 || rotation == 3
214  tmp = x;
215  x = y;
216  y = tmp;
217  }
218  }
219 
220  if (x < 0 || x >= WIDTH || y < 0 || y >= HEIGHT) {
221  // Ignore pixels outside the canvas.
222  return;
223  }
224 
225  // Translate the x, y coordinate according to the layout of the
226  // displays. They can be ordered and rotated (0, 90, 180, 270).
227 
228  uint8_t display = matrixPosition[(x >> 3) + hDisplays * (y >> 3)];
229  x &= 0b111;
230  y &= 0b111;
231 
232  uint8_t r = matrixRotation[display];
233  if (r >= 2) {
234  // 180 or 270 degrees
235  x = 7 - x;
236  }
237  if (r == 1 || r == 2) {
238  // 90 or 180 degrees
239  y = 7 - y;
240  }
241  if (r & 1) {
242  // 90 or 270 degrees
243  tmp = x;
244  x = y;
245  y = tmp;
246  }
247 
248  uint8_t d = display / hDisplays;
249  x += (display - d * hDisplays) << 3; // x += (display % hDisplays) * 8
250  y += d << 3; // y += (display / hDisplays) * 8
251 
252  // Update the color bit in our bitmap buffer.
253 
254  uint8_t *ptr = bitmap + x + WIDTH * (y >> 3);
255  uint8_t val = 1 << (y & 0b111);
256 
257  if (color) {
258  *ptr |= val;
259  } else {
260  *ptr &= ~val;
261  }
262  }
